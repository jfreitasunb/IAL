\begin{exercicio}

\begin{exercicio}
    Sejam $U$ e $V$ espaços vetoriais sobre um mesmo corpo $\cp{K}$. Defina o \textbf{produto cartesiano} de $U$ e $V$ como sendo
    \[
        U \times V = \{(x, y) \mid x \in U, y \in V\}.
    \]
    Prove que $U \times V$ é um espaço vetorial sobre $\cp{K}$.
\end{exercicio}

\end{exercicio}
